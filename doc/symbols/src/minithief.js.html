 39 // BEHAVIOURS
 40 /**
 41 * @class
 42 * @extends {Script}
 43 * @classdesc This class represents a behaviour for a GameElement.
 44 * @param {Object} param The parameters of the behaviour.
 45 * The object param contains these callbacks:
 46 <ul>
 47 <li><strong>start</strong> : This function runs when the engine start at first time.</li>
 48 <li><strong>update</strong> : This function runs each frame.</li>
 49 <li><strong>onEnterCollision</strong> : This function runs when the game element collides with other game element.</li>
 50 <li><strong>onExitCollision</strong> : This function runs when the collision is finished.</li>
 51 <li><strong>onDestroy</strong> : This function runs when the game element is destroyed.</li>
 52 </ul>
 53 */
 54 var Behaviour = function (param) {
 55     Script.call(this);
 56 
 57     this.gameElement = null;
 58 
 59     if(param !== null){
 60       this._start = param.start || function(){};
 61       this._update = param.update || function(){};
 62       this._onEnterCollision = param.onEnterCollision || function(){};
 63       this._onExitCollision = param.onExitCollision || function(){};
 64       this._onDestroy = param.onDestroy || function(){};
 65     }
 66 };
 67 
 68 Behaviour.prototype = new Script();
 69 Behaviour.prototype.constructor = Behaviour;
 70 
 71 //----------------------------------------------------------------------
 72 
 73 Behaviour.prototype.start = function () {
 74   this._start();
 75   this.gameElement.body.getBox2dBody().SetFixedRotation(true);
 76 };
 77 
 78 //----------------------------------------------------------------------
 79 
 80 Behaviour.prototype.update = function (){
 81   this._update();
 82 };
 83 
 84 //----------------------------------------------------------------------
 85 
 86 Behaviour.prototype.onEnterCollision = function (otherGameObject, contact){
 87   this._onEnterCollision(otherGameObject.getName());
 88 };
 89 
 90 //----------------------------------------------------------------------
 91 
 92 Behaviour.prototype.onExitCollision = function (otherGameObject, contact){
 93   this._onExitCollision(otherGameObject.getName());
 94 };
 95 
 96 //----------------------------------------------------------------------
 97 
 98 Behaviour.prototype.onDestroy = function (){
 99   this._onDestroy();
100 };

129 /**
130 * @class
131 * @extends {Script}
132 * @classdesc This class represents a GameElement.
133 * @param {Object} param The parameters of the gameelement.
134 * The object param contains these fields:
135 <ul>
136 <li><strong>position</strong> : {Point} The position.</li>
137 <li><strong>texture</strong> : {String} The name of the texture.</li>
138 <li><strong>width</strong> : {Number} The width.</li>
139 <li><strong>height</strong> : {Number} The height.</li>
140 <li><strong>shape</strong> : {Shape} The shape.</li>
141 <li><strong>behaviour</strong> : {Object} The behaviour's parameters.</li>
142 <li><strong>isStatic</strong> : {Bool} True if static.</li>
143 <li><strong>isSensor</strong> : {Bool} True if sensor.</li>
144 <li><strong>name</strong> : {String} The name of the game element.</li>
145 </ul>
146 */
147 var GameElement = function (param) {
148 
149   if(param !== null){
150 
151     var position = param.position || new Point(0,0);
152     var texture = param.texture || null;
153     var width = param.width || 1;
154     var height = param.height || 1;
155     var shape = param.shape || Shape.RECTANGLE;
156     var behaviour = param.behaviour || {};
157     var isStatic = param.isStatic || false;
158     var isSensor = param.isSensor || false;
159     var name = param.name || null;
160 
161     this.spriteBuilder = new SpriteBuilder();
162 
163     this.spriteBuilder.create(texture,position,width,height).
164       setName(name).
165       setStatic(isStatic).
166       setRigidBody(1,0.5,0.5); // set physics properties
167 
168     var col = null;
169 
170     if(shape == Shape.RECTANGLE)
171       col = new AABBCollider(width,height,isSensor); // set a Rectangle Collider
172     else if(shape == Shape.CIRCLE)
173       col = new CircleCollider(Math.max(width,height)/2,isSensor); // set a Circle Collider
174 
175     this.spriteBuilder.setCollider(col);
176 
177     this.gameObject = this.spriteBuilder.end();
178 
179     if(behaviour !== null)
180       this.addBehaviour(behaviour);
181 
182     // get the transform and the physic body
183     this.body = this.gameObject.getComponent(RigidBody);
184     this.transform = this.gameObject.getTransform();
185   }
186 
187 };
188 
189 /**
190 * Returns the name of the game element.
191 * @returns {String} The name.
192 */
193 GameElement.prototype.getName = function () {
194   return this.gameObject.getName();
195 };
196 
197 /**
198 * Returns the position of the game element.
199 * @returns {Point} The name.
200 */
201 GameElement.prototype.getPosition = function () {
202   return this.transform.getPosition();
203 };
204 
205 /**
206 * Moves the game element to a position.
207 * @param {Point} position The new position.
208 */
209 GameElement.prototype.moveTo = function (position) {
210   this.body.getBox2dBody().SetPosition(new b2Vec2(position.x,position.y));
211 };
212 
213 /**
214 * Applies a force to the game element.
215 * @param {Point} vector The force vector.
216 */
217 GameElement.prototype.applyForce = function (vector) {
218   this.body.getBox2dBody().ApplyForce(new b2Vec2(vector.x,vector.y),this.body.GetWorldCenter());
219 };
220 
221 /**
222 * Sets the velocity of the game element.
223 * @param {Point} vector The velocity vector.
224 */
225 GameElement.prototype.setVelocity = function (vector) {
226 
227   var linear = this.body.getBox2dBody().GetLinearVelocity();
228 
229   linear.x = vector.x;
230   linear.y = vector.y;
231 };
232 
233 /**
234 * Rotates the game element.
235 * @param {number} angle The angle in degrees.
236 */
237 GameElement.prototype.rotate = function (angle) {
238 
239 };
240 
241 /**
242 * Adds a new behaviour to the game element.
243 * @param {Object} behaviourParam The behaviour's parameters.
244 */
245 GameElement.prototype.addBehaviour = function (behaviourParam) {
246   var b = new Behaviour(behaviourParam);
247   b.gameElement = this;
248   this.gameObject.addComponent(b);
249 };
250 
251 /**
252 * Destroys the game element.
253 */
254 GameElement.prototype.destroy = function () {
255   this.gameObject.destroy();
256 };

272 * Initializes the engine.
273 */
274 Game.init = function () {
275   // inicializa motor
276 
277   Thief.init();
278   Thief.createAndSetScene("test");
279 
280   // camera
281 
282   var canvas = document.getElementById("glcanvas");
283   var screenW = canvas.width;
284   var screenH = canvas.height;
285   var zoom = 5;
286   var aspect = (screenW/screenH);
287   var w = 1*aspect;
288   var h = 1;
289 
290 
291   var camBuilder = new CameraBuilder();
292 
293   var cam =
294   camBuilder.create(new Vector3(0,0,15)).
295     setOrtho(w*zoom,h*zoom, -100,100).
296   end();
297 
298   Thief.addGameObjectToScene(cam);
299   Thief.setCamera(cam);
300 };
