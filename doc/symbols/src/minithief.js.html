  7 /**
  8 * @class
  9 * @extends {Vector2}
 10 * @classdesc This class represents a point (x,y) into an 2D space
 11 * @param {Number} x The x component.
 12 * @param {Number} y The y component.
 13 */
 14 var Point = function (x,y){
 15     Vector2.call(this,x,y);
 16 };
 17 
 18 Point.prototype = new Vector2();
 19 Point.prototype.constructor = Point;

 34 // BEHAVIOURS
 35 /**
 36 * @class
 37 * @extends {Script}
 38 * @classdesc This class represents a behaviour for a GameElement.
 39 * @param {Object} param The parameters of the behaviour.
 40 */
 41 var Behaviour = function (param) {
 42     Script.call(this);
 43 
 44     this.gameElement = null;
 45 
 46     if(param !== null){
 47       this._start = param.start || function(){};
 48       this._update = param.update || function(){};
 49       this._onEnterCollision = param.onEnterCollision || function(){};
 50       this._onExitCollision = param.onExitCollision || function(){};
 51       this._onDestroy = param.onDestroy || function(){};
 52     }
 53 };
 54 
 55 Behaviour.prototype = new Script();
 56 Behaviour.prototype.constructor = Behaviour;
 57 
 58 //----------------------------------------------------------------------
 59 
 60 Behaviour.prototype.start = function () {
 61   this._start();
 62   this.gameElement.body.getBox2dBody().SetFixedRotation(true);
 63 };
 64 
 65 //----------------------------------------------------------------------
 66 
 67 Behaviour.prototype.update = function (){
 68   this._update();
 69 };

 75 };

 95     var v = 2;

 98       // LEFT

100     }else if (Input.isKeyPressed(38)) {
101       // UP

103     }else if (Input.isKeyPressed(39)) {
104       // RIGHT
105       this.gameElement.setVelocity(new Vector2(v,0));
106     }else if (Input.isKeyPressed(40)) {
107       // DOWN
108       this.gameElement.setVelocity(new Vector2(0,-v));
109     }
110   }
111 };

116 /**
117 * @class
118 * @extends {Script}
119 * @classdesc This class represents a GameElement.
120 * @param {Object} param The parameters of the gameelement.
121 */
122 var GameElement = function (param) {
123 
124   if(param !== null){
125 
126     var position = param.position || new Point(0,0);
127     var texture = param.texture || null;
128     var width = param.width || 1;
129     var height = param.height || 1;
130     var shape = param.shape || Shape.RECTANGLE;
131     var behaviour = param.behaviour || {};
132     var isStatic = param.isStatic || false;
133     var isSensor = param.isSensor || false;
134     var name = param.name || null;
135 
136     this.spriteBuilder = new SpriteBuilder();
137 
138     this.spriteBuilder.create(texture,position,width,height).
139       setName(name).
140       setStatic(isStatic).
141       setRigidBody(1,0.5,0.5); // set physics properties
142 
143     var col = null;
144 
145     if(shape == Shape.RECTANGLE)
146       col = new AABBCollider(width,height,isSensor); // set a Rectangle Collider
147     else if(shape == Shape.CIRCLE)
148       col = new CircleCollider(Math.max(width,height)/2,isSensor); // set a Circle Collider
149 
150     this.spriteBuilder.setCollider(col);
151 
152     this.gameObject = this.spriteBuilder.end();
153 
154     if(behaviour !== null)
155       this.addBehaviour(behaviour);
156 
157     // get the transform and the physic body
158     this.body = this.gameObject.getComponent(RigidBody);
159     this.transform = this.gameObject.getTransform();
160   }
161 
162 };
163 
164 GameElement.prototype.getName = function () {
165   return this.gameObject.getName();
166 };
167 
168 GameElement.prototype.getPosition = function () {
169   return this.transform.getPosition();
170 };
171 
172 GameElement.prototype.moveTo = function (position) {
173   this.body.getBox2dBody().SetPosition(new b2Vec2(position.x,position.y));
174 };
175 
176 GameElement.prototype.applyForce = function (vector) {
177   this.body.getBox2dBody().ApplyForce(new b2Vec2(vector.x,vector.y),this.body.GetWorldCenter());
178 };
179 
180 GameElement.prototype.setVelocity = function (vector) {
181 
182   var linear = this.body.getBox2dBody().GetLinearVelocity();
183 
184   linear.x = vector.x;
185   linear.y = vector.y;
186 };
187 
188 GameElement.prototype.rotate = function () {
189 
190 };
191 
192 GameElement.prototype.addBehaviour = function (behaviourParam) {
193   var b = new Behaviour(behaviourParam);
194   b.gameElement = this;
195   this.gameObject.addComponent(b);
196 };
197 
198 GameElement.prototype.destroy = function () {
199   this.gameObject.destroy();
200 };

212 Game.init = function () {
213   // inicializa motor
214 
215   Thief.init();
216   Thief.createAndSetScene("test");
217 
218   // camera
219 
220   var canvas = document.getElementById("glcanvas");
221   var screenW = canvas.width;
222   var screenH = canvas.height;
223   var zoom = 5;
224   var aspect = (screenW/screenH);
225   var w = 1*aspect;
226   var h = 1;
227 
228 
229   var camBuilder = new CameraBuilder();
230 
231   var cam =
232   camBuilder.create(new Vector3(0,0,15)).
233     setOrtho(w*zoom,h*zoom, -100,100).
234   end();
235 
236   Thief.addGameObjectToScene(cam);
237   Thief.setCamera(cam);
238 };
